73  //コマンドを処理する
74  bool HandleCommand(CEditView* View, EFunctionCode ID, const WCHAR* Arguments[], const int ArgLengths[], const int ArgSize)
75  {
76  switch ( LOWORD(ID) )
77  {
78  case F_OL_ADDFUNCINFO: //アウトライン解析に追加する
79  case F_OL_ADDFUNCINFO2: //アウトライン解析に追加する（深さ指定）
80  case F_OL_ADDFUNCINFO3: //アウトライン解析に追加する（ファイル名）
81  case F_OL_ADDFUNCINFO4: //アウトライン解析に追加する（ファイル名/深さ指定）
82  {
83  if( Arguments[0] == NULL )return false;
84  if( Arguments[1] == NULL )return false;
85  if( Arguments[2] == NULL )return false;
86  if( Arguments[3] == NULL )return false;
87  CLogicPoint ptLogic( _wtoi(Arguments[1])-1, _wtoi(Arguments[0])-1 );
88  CLayoutPoint ptLayout;
89  if( ptLogic.x < 0 || ptLogic.y < 0 ){
90  ptLayout.x = (Int)ptLogic.x;
91  ptLayout.y = (Int)ptLogic.y;
92  }else{
93  View->GetDocument()->m_cLayoutMgr.LogicToLayout( ptLogic, &ptLayout );
94  }
95  int nParam = _wtoi(Arguments[3]);
96  if( LOWORD(ID) == F_OL_ADDFUNCINFO ){
97  m_cFuncInfoArr.AppendData( ptLogic.GetY()+1, ptLogic.GetX()+1, ptLayout.GetY()+1, ptLayout.GetX()+1, Arguments[2], NULL, nParam );
98  }else if( LOWORD(ID) == F_OL_ADDFUNCINFO2 ){
99  int nDepth = nParam & FUNCINFO_INFOMASK;
100  nParam -= nDepth;
101  m_cFuncInfoArr.AppendData( ptLogic.GetY()+1, ptLogic.GetX()+1, ptLayout.GetY()+1, ptLayout.GetX()+1, Arguments[2], NULL, nParam, nDepth );
102  }else if( LOWORD(ID) == F_OL_ADDFUNCINFO3 ){
103  if( ArgSize < 5 || Arguments[4] == NULL ){ return false; }
104  m_cFuncInfoArr.AppendData( ptLogic.GetY()+1, ptLogic.GetX()+1, ptLayout.GetY()+1, ptLayout.GetX()+1, Arguments[2], Arguments[4], nParam );
105  }else if( LOWORD(ID) == F_OL_ADDFUNCINFO4 ){
106  if( ArgSize < 5 || Arguments[4] == NULL ){ return false; }
107  int nDepth = nParam & FUNCINFO_INFOMASK;
108  nParam -= nDepth;
109  m_cFuncInfoArr.AppendData( ptLogic.GetY()+1, ptLogic.GetX()+1, ptLayout.GetY()+1, ptLayout.GetX()+1, Arguments[2], Arguments[4], nParam, nDepth );
110  }
111  }
112  break;
113  case F_OL_SETTITLE: //アウトラインダイアログタイトルを指定
114  if( Arguments[0] == NULL )return false;
115  m_sOutlineTitle = Arguments[0];
116  break;
117  case F_OL_SETLISTTYPE: //アウトラインリスト種別を指定
118  if( Arguments[0] == NULL )return false;
119  m_nListType = (EOutlineType)_wtol(Arguments[0]);
120  break;
121  case F_OL_SETLABEL: //ラベル文字列を指定
122  if( Arguments[0] == NULL || Arguments[1] == NULL ) return false;
123  {
124  std::wstring sLabel = Arguments[1];
125  m_cFuncInfoArr.SetAppendText( _wtol(Arguments[0]), sLabel, true );
126  }
127  break;
128  default:
129  return false;
130  }
131  return true;
132  }
