108  int Enumerates( LPCWSTR lpBaseFolder, VGrepEnumKeys& vecKeys, CGrepEnumOptions& option, CGrepEnumFileBase* pExceptItems = NULL ){
109  int found = 0;
110 
111  for( int i = 0; i < (int)vecKeys.size(); i++ ){
112  int baseLen = wcslen( lpBaseFolder );
113  LPWSTR lpPath = new WCHAR[ baseLen + wcslen( vecKeys[ i ] ) + 2 ];
114  if( NULL == lpPath ) break;
115  wcscpy( lpPath, lpBaseFolder );
116  wcscpy( lpPath + baseLen, L"\\" );
117  wcscpy( lpPath + baseLen + 1, vecKeys[ i ] );
118  // vecKeys[ i ] ==> "subdir\*.h" 等の場合に後で(ファイル|フォルダ)名に "subdir\" を連結する
119  const WCHAR* keyDirYen = wcsrchr( vecKeys[ i ], L'\\' );
120  const WCHAR* keyDirSlash = wcsrchr( vecKeys[ i ], L'/' );
121  const WCHAR* keyDir;
122  if( keyDirYen == NULL ){
123  keyDir = keyDirSlash;
124  }else if( keyDirSlash == NULL ){
125  keyDir = keyDirYen;
126  }else if( keyDirYen < keyDirSlash ){
127  keyDir = keyDirSlash;
128  }else{
129  keyDir = keyDirYen;
130  }
131  int nKeyDirLen = keyDir ? keyDir - vecKeys[ i ] + 1 : 0;
132 
133  WIN32_FIND_DATA w32fd;
134  HANDLE handle = ::FindFirstFile( lpPath, &w32fd );
135  if( INVALID_HANDLE_VALUE != handle ){
136  do{
137  if( !::PathMatchSpec(w32fd.cFileName, vecKeys[ i ] + nKeyDirLen) ){
138  continue;
139  }
140  if( option.m_bIgnoreHidden && (w32fd.dwFileAttributes & FILE_ATTRIBUTE_HIDDEN) ){
141  continue;
142  }
143  if( option.m_bIgnoreReadOnly && (w32fd.dwFileAttributes & FILE_ATTRIBUTE_READONLY) ){
144  continue;
145  }
146  if( option.m_bIgnoreSystem && (w32fd.dwFileAttributes & FILE_ATTRIBUTE_SYSTEM) ){
147  continue;
148  }
149  LPWSTR lpName = new WCHAR[ nKeyDirLen + wcslen( w32fd.cFileName ) + 1 ];
150  wcsncpy( lpName, vecKeys[ i ], nKeyDirLen );
151  wcscpy( lpName + nKeyDirLen, w32fd.cFileName );
152  LPWSTR lpFullPath = new WCHAR[ baseLen + wcslen(lpName) + 2 ];
153  wcscpy( lpFullPath, lpBaseFolder );
154  wcscpy( lpFullPath + baseLen, L"\\" );
155  wcscpy( lpFullPath + baseLen + 1, lpName );
156  if( IsValid( w32fd, lpName ) ){
157  if( pExceptItems && pExceptItems->IsExist( lpFullPath ) ){
158  }else{
159  m_vpItems.push_back( PairGrepEnumItem( lpName, w32fd.nFileSizeLow ) );
160  found++; // 2011.11.19
161  if( pExceptItems && nKeyDirLen ){
162  // フォルダを含んだパスなら検索済みとして除外指定に追加する
163  pExceptItems->m_vpItems.push_back( PairGrepEnumItem( lpFullPath, w32fd.nFileSizeLow ) );
164  }else{
165  delete [] lpFullPath;
166  }
167  continue;
168  }
169  }
170  delete [] lpName;
171  delete [] lpFullPath;
172  }while( ::FindNextFile( handle, &w32fd ) );
173  ::FindClose( handle );
174  }
175  delete [] lpPath;
176  }
177  return found;
178  }
179 };
