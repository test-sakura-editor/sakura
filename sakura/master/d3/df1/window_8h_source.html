49  static int ScaleX(int x){Init(); return ::MulDiv(x, nDpiX, 96);}
50  static int ScaleY(int y){Init(); return ::MulDiv(y, nDpiY, 96);}
51  static int UnscaleX(int x){Init(); return ::MulDiv(x, 96, nDpiX);}
52  static int UnscaleY(int y){Init(); return ::MulDiv(y, 96, nDpiY);}
53  static void ScaleRect(LPRECT lprc)
54  {
55  lprc->left = ScaleX(lprc->left);
56  lprc->right = ScaleX(lprc->right);
57  lprc->top = ScaleY(lprc->top);
58  lprc->bottom = ScaleY(lprc->bottom);
59  }
60  static void UnscaleRect(LPRECT lprc)
61  {
62  lprc->left = UnscaleX(lprc->left);
63  lprc->right = UnscaleX(lprc->right);
64  lprc->top = UnscaleY(lprc->top);
65  lprc->bottom = UnscaleY(lprc->bottom);
66  }
67  static int PointsToPixels(int pt, int ptMag = 1){Init(); return ::MulDiv(pt, nDpiY, 72 * ptMag);} // ptMag: 引数のポイント数にかかっている倍率
68  static int PixelsToPoints(int px, int ptMag = 1){Init(); return ::MulDiv(px * ptMag, 72, nDpiY);} // ptMag: 戻り値のポイント数にかける倍率
69 };
