30 /*
31  2007.10.18 kobake
32  テンプレート式 min とか max とか。
33 
34  どっかの標準ヘッダに、同じようなものがあった気がするけど、
35  NOMINMAX を定義するにしても、なんだか min とか max とかいう名前だと、
36  テンプレートを呼んでるんだかマクロを呼んでるんだか訳分かんないので、
37  明示的に「t_～」という名前を持つ関数を用意。
38 */
39 
40 template <class T>
41 inline T t_min(T t1,T t2)
42 {
43  return t1<t2?t1:t2;
44 }
45 
46 template <class T>
47 inline T t_max(T t1,T t2)
48 {
49  return t1>t2?t1:t2;
50 }
51 
52 template <class T>
53 T t_abs(T t)
54 {
55  return t>=T(0)?t:T(-t);
56 }
57 
58 template <class T>
59 T t_unit(T t)
60 {
61  return
62  t>T(0)?1:
63  t<T(0)?-1:
64  0;
65 }
