66 inline int CShiftJis::_SjisToUni_char( const unsigned char *pSrc, unsigned short *pDst, const ECharSet eCharset, bool* pbError )
67 {
68  int nret;
69  bool berror = false;
70 
71  switch( eCharset ){
72  case CHARSET_JIS_HANKATA:
73  // 半角カタカナを処理
74  // エラーは起こらない。
75  nret = MyMultiByteToWideChar_JP( pSrc, 1, pDst );
76  // 保護コード
77  if( nret < 1 ){
78  nret = 1;
79  }
80  break;
81  case CHARSET_JIS_ZENKAKU:
82  // 全角文字を処理
83  nret = MyMultiByteToWideChar_JP( pSrc, 2, pDst );
84  if( nret < 1 ){ // SJIS -> Unicode 変換に失敗
85  nret = BinToText( pSrc, 2, pDst );
86  }
87  break;
88  default:
89  // 致命的エラー回避コード
90  berror = true;
91  pDst[0] = L'?';
92  nret = 1;
93  }
94 
95  if( pbError ){
96  *pbError = berror;
97  }
98 
99  return nret;
100 }
