115  // 2010/4/19 Uchi
116  // 2011/8/20 syat 関数コードの割り当て直し
117  static inline PluginId GetPluginId( EFunctionCode nFunctionCode )
118  {
119  if (nFunctionCode >= F_PLUGCOMMAND_FIRST && nFunctionCode < F_PLUGCOMMAND_LAST) {
120  return PluginId( (nFunctionCode - F_PLUGCOMMAND_FIRST)/100 + (nFunctionCode%100/50 * 20) );
121  }
122  return PluginId(-1);
123  }

125  // PluginNo番号の計算(クラス外でも使えるバージョン)
126  // 2010/6/24 Uchi

128  static inline PlugId GetPlugId( EFunctionCode nFunctionCode )
129  {
130  if (nFunctionCode >= F_PLUGCOMMAND_FIRST && nFunctionCode < F_PLUGCOMMAND_LAST) {
131  return PlugId( nFunctionCode%100 - (nFunctionCode%100/50 * 50) );
132  }
133  return PlugId(-1);

136  /* PluginId, PlugId と 関数コードのマッピング *****************************
137  * PluginId … プラグインの番号 0～39
138  * PlugId … プラグイン内のプラグの番号 0～49
139  *
140  * 関数コード 20000～21999 ()内は(PluginId, PlugId)を表す
141  * +------------+------------+----+------------+
142  * |20000(0,0) |20100(1,0) | |21900(19,0) |
143  * | : | : | … | : |
144  * |20049(0,49) |20149(1,49) | |21949(19,49)|
145  * +------------+------------+----+------------+
146  * |20050(20,0) |20150(21,0) | |21950(39,0) |
147  * | : | : | … | : |
148  * |20099(20,49)|20199(21,49)| |21999(39,49)|
149  * +------------+------------+----+------------+
150  * もし足りなければ、22000～23999を払い出して食いつぶす
