234 inline
235 ECodeType CESI::DetectUnicodeBom(const char* buff, size_t size) noexcept
236 {
237  // バイト列がない、または、BOM表現を格納できるサイズに満たない場合、判定をスキップ
238  if (!buff || size < 2) return CODE_NONE;
239 
240  // バイト列の先頭が \ufeff の utf8 表現と一致するか判定
241  constexpr const BYTE utf8BOM[]{ 0xef, 0xbb, 0xbf };
242  if (size >= _countof(utf8BOM) && 0 == ::memcmp(buff, utf8BOM, _countof(utf8BOM))) {
243  return CODE_UTF8;
244  }
245 
246  // バイト列の先頭が \ufeff の utf16BE 表現と一致するか判定
247  constexpr const BYTE utf16BeBOM[]{ 0xfe, 0xff };
248  if (size >= _countof(utf16BeBOM) && 0 == ::memcmp(buff, utf16BeBOM, _countof(utf16BeBOM))) {
249  return CODE_UNICODEBE;
250  }
251 
252  // バイト列の先頭が \ufeff の utf16LE 表現と一致するか判定
253  constexpr const BYTE utf16LeBOM[]{ 0xff, 0xfe };
254  if (size >= _countof(utf16LeBOM) && 0 == ::memcmp(buff, utf16LeBOM, _countof(utf16LeBOM))) {
255  return CODE_UNICODE;
256  }
257 
258  // UTF-7 は ASCII 7bit 文字 でない文字を UTF-16BE で符号化してから 修正BASE64 で 符号化する。
259  // Base64 の符号化は 6bit単位 なので BOM に続く文字が非7bit文字な場合、4バイト目がブレる。
260  // このため、 UTF-7 については BOM による判別ロジック省略の対象から外している。
261  //
262  // (BOM)abc ⇒ (UTF-7変換) ⇒ +/v8-abc
263  // (BOM)ｱｲｳ ⇒ (UTF-7変換) ⇒ +/v//cf9y/3M-
264 
265  return CODE_NONE;
266 }
