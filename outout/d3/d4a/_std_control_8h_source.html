27 /*
28 2007.09.17 kobake
29 内部コードがWCHARなので、検索キーワードなどもWCHARで保持する。
30 そのため、検索ダイアログのコンボボックスなどに、WCHARを設定する場面が出てくる。
31 UNICODE版では問題無いが、ANSI版では設定の前にコード変換する必要がある。
32 呼び出し側で変換しても良いが、頻度が多いので、WCHARを直接受け取るAPIラップ関数を提供する。
33 
34 また、SendMessageの直接呼び出しは、どうしてもWPARAM,LPARAMへの強制キャストが生じるため、
35 コンパイラの型チェックが働かず、wchar_t, charの混在するソースコードの中ではバグの温床になりやすい。
36 そういった意味でも、このファイル内のラップ関数を使うことを推奨する。
37 */

71  inline bool Wnd_GetText(HWND hwnd, CNativeW& str)
72  {
73  // バッファをクリアしておく
74  str.Clear();
75 
76  // GetWindowTextLength() はウィンドウテキスト取得に必要なバッファサイズを返す。
77  // 条件によっては必要なサイズより大きな値を返すことがある模様
78  // https://docs.microsoft.com/en-us/windows/desktop/api/winuser/nf-winuser-getwindowtextlengtha
79  const int length = ::GetWindowTextLength(hwnd);
80  if (length < 0)
81  {
82  // ドキュメントには失敗した場合、あるいはテキストが空の場合には 0 を返すとある。
83  // 0 の場合はエラーかどうか判断できないのでテキストの取得処理を続行する。
84  // 仕様上は負の場合はありえないが、念の為エラーチェックしておく。
85  return false;
86  }
87 
88  // ドキュメントには NULL 文字に関して言及がないので念の為 +1 しておく
89  const int bufsize = length + 1;
90 
91  // ウィンドウタイトルを設定するのに必要なバッファを確保する
92  if (str.capacity() < bufsize)
93  {
94  str.AllocStringBuffer(bufsize);
95  }
96 
97  int actualCount = ::GetWindowText(hwnd, str.GetStringPtr(), str.capacity());
98  if (actualCount < 0)
99  {
100  // 仕様上は負の場合はありえないが、念の為エラーチェックしておく。
101  return false;
102  }
103  else if (actualCount == 0)
104  {
105  // GetWindowText はエラーの場合、またはテキストが空の場合は 0 を返す
106  if (GetLastError() != 0)
107  {
108  return false;
109  }
110  }
111  else if(actualCount >= str.capacity())
112  {
113  // GetWindowText() の仕様上はありえないはず
114  return false;
115  }
116 
117  // Win32 API の GetWindowText() を呼んだだけでは CNativeW 内部の
118  // データサイズが更新されないのでデータサイズを反映する
119  str._SetStringLength(actualCount);
120 
121  // 正しく設定されているはず
122  assert(str.GetStringLength() == actualCount);
123  return true;
124  }
