42 inline bool MyFillRect( const HDC hDC, const RECT &rc, const HBRUSH hBrush ) noexcept
43 {
44  assert( hDC );
45  assert( hBrush );
46 
47  if ( !hDC || !hBrush ) return false;
48 
49  HGDIOBJ hBrushOld = ::SelectObject( hDC, hBrush );
50  if ( !hBrushOld || hBrushOld == HGDI_ERROR ) return false;
51 
52  auto retPatBlt = ::PatBlt( hDC, rc.left, rc.top, rc.right - rc.left, rc.bottom - rc.top, PATCOPY );
53  ::SelectObject( hDC, hBrushOld );
54 
55  return retPatBlt != 0;
56 }
