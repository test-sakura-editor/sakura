35 class CCaretUnderLine
36 {
37 public:
38  CCaretUnderLine(CEditView* pcEditView)
39  : m_pcEditView(pcEditView)
40  {
41  m_nLockCounter = 0;
42  m_nUnderLineLockCounter = 0;
43  }
44  // 表示非表示を切り替えられないようにする
45  void Lock()
46  {
47  m_nLockCounter++;
48  }
49  // 表示非表示を切り替えられるようにする
50  void UnLock()
51  {
52  m_nLockCounter--;
53  if (m_nLockCounter < 0){
54  m_nLockCounter = 0;
55  }
56  }
57  void UnderLineLock()
58  {
59  m_nUnderLineLockCounter++;
60  }
61  // 表示非表示を切り替えられるようにする
62  void UnderLineUnLock()
63  {
64  m_nUnderLineLockCounter--;
65  if (m_nUnderLineLockCounter < 0){
66  m_nUnderLineLockCounter = 0;
67  }
68  }
69  void CaretUnderLineON(bool bDraw, bool bPaintDraw); // カーソル行アンダーラインのON
70  void CaretUnderLineOFF(bool bDraw, bool bDrawPaint = true, bool bResetFlag = false ); // カーソル行アンダーラインのOFF
71  void SetUnderLineDoNotOFF( bool flag ){ if( !m_nLockCounter )m_bUnderLineDoNotOFF = flag; }
72  void SetVertLineDoNotOFF( bool flag ){ if( !m_nLockCounter )m_bVertLineDoNotOFF = flag; }
73  inline bool GetUnderLineDoNotOFF( )const { return m_bUnderLineDoNotOFF; }
74  inline bool GetVertLineDoNotOFF( )const { return m_bVertLineDoNotOFF; }
75 private:
76  /* ロックカウンタ。0のときは、ロックされていない。UnLockが呼ばれすぎても負にはならない */
77  int m_nLockCounter;
78  int m_nUnderLineLockCounter;
79  CEditView* m_pcEditView;
80  bool m_bUnderLineDoNotOFF;
81  bool m_bVertLineDoNotOFF;
82 };
