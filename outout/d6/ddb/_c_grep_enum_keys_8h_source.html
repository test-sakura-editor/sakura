166  static std::vector< wstring > SplitPattern(LPCWSTR lpKeys)
167  {
168  std::vector< wstring > patterns;
169 
170  const WCHAR* WILDCARD_DELIMITER = L" ;,"; //リストの区切り
171  int nWildCardLen = wcslen(lpKeys);
172  WCHAR* pWildCard = new WCHAR[nWildCardLen + 1];
173  if (!pWildCard) {
174  return patterns;
175  }
176  wcscpy(pWildCard, lpKeys);
177 
178  int nPos = 0;
179  WCHAR* token;
180  while (NULL != (token = my_strtok<WCHAR>(pWildCard, nWildCardLen, &nPos, WILDCARD_DELIMITER))) { //トークン毎に繰り返す。
181  // "を取り除いて左に詰める
182  WCHAR* p;
183  WCHAR* q;
184  p = q = token;
185  while (*p) {
186  if (*p != L'"') {
187  if (p != q) {
188  *q = *p;
189  }
190  q++;
191  }
192  p++;
193  }
194  *q = L'\0';
195 
196  wstring element(token);
197  patterns.push_back(element);
198  }
199  delete[] pWildCard;
200  return patterns;
201  }
