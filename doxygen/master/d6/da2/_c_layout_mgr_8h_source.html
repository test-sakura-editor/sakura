161  CLayoutInt GetActualTsvSpace(CLayoutInt pos, wchar_t ch) const {
162  CLayoutInt tabPadding = m_nCharLayoutXPerKeta - 1;
163  if (m_tsvInfo.m_nTsvMode == TSV_MODE_NONE && ch == WCODE::TAB) {
164  return GetTabSpace() + tabPadding - ((pos + tabPadding) % GetTabSpace());
165  }
166  else if (m_tsvInfo.m_nTsvMode == TSV_MODE_CSV && ch == WCODE::TAB) {
167  return tabPadding;
168  }
169  else if ((m_tsvInfo.m_nTsvMode == TSV_MODE_TSV && ch == WCODE::TAB)
170  || (m_tsvInfo.m_nTsvMode == TSV_MODE_CSV && ch == L',')) {
171  return CLayoutInt(m_tsvInfo.GetActualTabLength(pos, m_tsvInfo.m_nMaxCharLayoutX));
172  }
173  else {
174  return tabPadding;
