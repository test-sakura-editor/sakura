37 struct MYDEVMODE {
38  BOOL m_bPrinterNotFound; /* プリンタがなかったフラグ */
39  WCHAR m_szPrinterDriverName[_MAX_PATH + 1]; // プリンタドライバ名
40  WCHAR m_szPrinterDeviceName[_MAX_PATH + 1]; // プリンタデバイス名
41  WCHAR m_szPrinterOutputName[_MAX_PATH + 1]; // プリンタポート名
42  DWORD dmFields;
43  short dmOrientation;
44  short dmPaperSize;
45  short dmPaperLength;
46  short dmPaperWidth;
47 
49  bool operator == (const MYDEVMODE& rhs) const noexcept {
50  if (this == &rhs) return true;
51  return m_bPrinterNotFound == rhs.m_bPrinterNotFound
52  && 0 == wcsncmp(m_szPrinterDriverName, rhs.m_szPrinterDriverName, _countof(m_szPrinterDriverName))
53  && 0 == wcsncmp(m_szPrinterDeviceName, rhs.m_szPrinterDeviceName, _countof(m_szPrinterDeviceName))
54  && 0 == wcsncmp(m_szPrinterOutputName, rhs.m_szPrinterOutputName, _countof(m_szPrinterOutputName))
55  && dmFields == rhs.dmFields
56  && dmOrientation == rhs.dmOrientation
57  && dmPaperSize == rhs.dmPaperSize
58  && dmPaperLength == rhs.dmPaperLength
59  && dmPaperWidth == rhs.dmPaperWidth
60  ;
61  }
63  bool operator != (const MYDEVMODE& rhs) const noexcept {
64  return !(*this == rhs);
65  }
66 };
