62 inline int CLatin1::_UniToLatin1_char( const unsigned short* pSrc, unsigned char* pDst, const ECharSet eCharset, bool* pbError )
63 {
64  int nret;
65  bool berror = false;
66  BOOL blost;
67 
68  if( eCharset == CHARSET_UNI_NORMAL ){
69  if ((pSrc[0] >= 0 && pSrc[0] <= 0x7f) || (pSrc[0] >= 0xa0 && pSrc[0] <= 0xff)) {
70  // ISO 58859-1の範囲
71  pDst[0] = (unsigned char)pSrc[0];
72  nret = 1;
73  } else {
74  // ISO 8859-1以外
75  nret = ::WideCharToMultiByte( 1252, 0, reinterpret_cast<const wchar_t*>(pSrc), 1, reinterpret_cast<char*>(pDst), 4, NULL, &blost );
76  if( blost != FALSE ){
77  // Uni -> CLatin1 変換に失敗
78  berror = true;
79  pDst[0] = '?';
80  nret = 1;
81  }
82  }
83  }else if( eCharset == CHARSET_UNI_SURROG ){
84  // サロゲートペアは CLatin1 に変換できない。
85  berror = true;
86  pDst[0] = '?';
87  nret = 1;
88  }else{
89  // 保護コード
90  berror = true;
91  pDst[0] = '?';
92  nret = 1;
93  }
94

96  *pbError = berror;
97  }
98
