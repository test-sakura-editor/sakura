81  //関数を処理する
82  bool HandleFunction(CEditView* View, EFunctionCode ID, const VARIANT *Arguments, const int ArgSize, VARIANT &Result)
83  {
84  Variant varCopy; // VT_BYREFだと困るのでコピー用
85 
86  switch(LOWORD(ID))
87  {
88  case F_PL_GETPLUGINDIR: //プラグインフォルダパスを取得する
89  {
90  SysString S(m_cPlugin.m_sBaseDir.c_str(), m_cPlugin.m_sBaseDir.size());
91  Wrap(&Result)->Receive(S);
92  }
93  return true;
94  case F_PL_GETDEF: //設定ファイルから値を読む
95  case F_PL_GETOPTION: //オプションファイルから値を読む
96  {
97  CDataProfile cProfile;
98  wstring sSection;
99  wstring sKey;
100  wstring sValue;
101  if( variant_to_wstr( Arguments[0], sSection ) != true) return false;
102  if( variant_to_wstr( Arguments[1], sKey ) != true) return false;
103 
104  cProfile.SetReadingMode();
105  if( LOWORD(ID) == F_PL_GETDEF ){
106  cProfile.ReadProfile( m_cPlugin.GetPluginDefPath().c_str() );
107  }else{
108  cProfile.ReadProfile( m_cPlugin.GetOptionPath().c_str() );
109  }
110  if (!cProfile.IOProfileData( sSection.c_str(), sKey.c_str(), sValue )
111  && LOWORD(ID) == F_PL_GETOPTION ) {
112  // 設定されていなければデフォルトを取得
113  CPluginOption::ArrayIter it;
114  for (it = m_cPlugin.m_options.begin(); it != m_cPlugin.m_options.end(); it++) {
115  wstring sSectionTmp;
116  wstring sKeyTmp;
117  (*it)->GetKey(&sSectionTmp, &sKeyTmp);
118  if (sSection == sSectionTmp && sKey == sKeyTmp) {
119  sValue = (*it)->GetDefaultVal();
120  break;
121  }
122  }
123  }
124 
125  SysString S(sValue.c_str(), sValue.size());
126  Wrap(&Result)->Receive(S);
127  }
128  return true;
129  case F_PL_GETCOMMANDNO: //実行中プラグの番号を取得する
130  {
131  Wrap(&Result)->Receive(m_nPlugIndex);
132  }
133  return true;
134  case F_PL_GETSTRING:
135  {
136  int num;
137  if(variant_to_int( Arguments[0], num ) == false) return false;
138  if( 0 < num && num < MAX_PLUG_STRING ){
139  std::wstring& str = m_cPlugin.m_aStrings[num];
140  SysString S(str.c_str(), str.size());
141  Wrap(&Result)->Receive(S);
142  return true;
143  }else if( 0 == num ){
144  std::wstring str = m_cPlugin.m_sLangName.c_str();
145  SysString S(str.c_str(), str.size());
146  Wrap(&Result)->Receive(S);
147  return true;
148  }
149  }
150  }
151  return false;
152  }
153  //コマンドを処理する
154  bool HandleCommand(CEditView* View, EFunctionCode ID, const WCHAR* Arguments[], const int ArgLengths[], const int ArgSize)
155  {
156  switch ( LOWORD(ID) )
157  {
158  case F_PL_SETOPTION: //オプションファイルに値を書く
159  {
160  if( Arguments[0] == NULL )return false;
161  if( Arguments[1] == NULL )return false;
162  if( Arguments[2] == NULL )return false;
163  CDataProfile cProfile;
164 
165  cProfile.ReadProfile( m_cPlugin.GetOptionPath().c_str() );
166  cProfile.SetWritingMode();
167  wstring tmp(Arguments[2]);
168  cProfile.IOProfileData( Arguments[0], Arguments[1], tmp );
169  cProfile.WriteProfile( m_cPlugin.GetOptionPath().c_str(), (m_cPlugin.m_sName + L" プラグイン設定ファイル").c_str() );
170  }
171  break;
172  case F_PL_ADDCOMMAND: //コマンドを追加する
173  {
174  int id = m_cPlugin.AddCommand( Arguments[0], Arguments[1], Arguments[2], true );
175  View->m_pcEditWnd->RegisterPluginCommand( id );
176  }
177  break;
178  }
179  return true;
180  }
