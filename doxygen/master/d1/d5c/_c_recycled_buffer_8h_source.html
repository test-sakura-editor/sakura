31 class CRecycledBuffer{
32 //コンフィグ
33 private:
34  static const int BLOCK_SIZE = 1024; //ブロックサイズ。バイト単位。
35  static const int CHAIN_COUNT = 64; //再利用可能なブロック数。
36 
37 //コンストラクタ・デストラクタ
38 public:
39  CRecycledBuffer()
40  {
41  m_current=0;
42  }
43 
44 //インターフェース
45 public:
47  template <class T>
48  T* GetBuffer(
49  size_t* nCount
50  )
51  {
52  if(nCount)*nCount=BLOCK_SIZE/sizeof(T);
53  m_current = (m_current+1) % CHAIN_COUNT;
54  return reinterpret_cast<T*>(m_buf[m_current]);
55  }
56 
58  template <class T>
59  size_t GetMaxCount() const
60  {
61  return BLOCK_SIZE/sizeof(T);
62  }
63 
64 //メンバ変数
65 private:
66  BYTE m_buf[CHAIN_COUNT][BLOCK_SIZE];
67  int m_current;
68 };

87  if(m_buf[i])delete[] m_buf[i];
88  }
89  }
90 
91 //インターフェース
92 public:
94  template <class T>
95  T* GetBuffer(
96  size_t nCount
97  )
98  {
99  m_current = (m_current+1) % CHAIN_COUNT;
100 
101  //メモリ確保
102  if(m_buf[m_current])delete[] m_buf[m_current];
103  m_buf[m_current]=new BYTE[nCount*sizeof(T)];
104 
105  return reinterpret_cast<T*>(m_buf[m_current]);
106  }
107 
108 //メンバ変数
109 private:
110  BYTE* m_buf[CHAIN_COUNT];
