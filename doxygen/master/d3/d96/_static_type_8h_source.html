31 template <class ELEMENT_TYPE, int MAX_SIZE, class SET_TYPE = const ELEMENT_TYPE&>
32 class StaticVector{
33 public:
34  //型
35  typedef ELEMENT_TYPE ElementType;
36 
37 public:
38  //属性
39  int size() const{ return m_nCount; }
40  int max_size() const{ return MAX_SIZE; }
41 
42  //要素アクセス
43  ElementType& operator[](int nIndex)
44  {
45  assert(nIndex<MAX_SIZE);
46  assert_warning(nIndex<m_nCount);
47  return m_aElements[nIndex];
48  }
49  const ElementType& operator[](int nIndex) const
50  {
51  assert(nIndex<MAX_SIZE);
52  assert_warning(nIndex<m_nCount);
53  return m_aElements[nIndex];
54  }
55 
56  //操作
57  void clear(){ m_nCount=0; }
58  void push_back(SET_TYPE e)
59  {
60  assert(m_nCount<MAX_SIZE);
61  m_nCount++;
62  m_aElements[m_nCount-1]=e;
63  }
64  void resize(int nNewSize)
65  {
66  assert(0 <= nNewSize);
67  assert(nNewSize <= MAX_SIZE);
68  m_nCount = nNewSize;
69  }
70 
72  ElementType* dataPtr(){ return m_aElements;}
73 
74  //特殊
75  int& _GetSizeRef(){ return m_nCount; }
76  void SetSizeLimit(){
77  if( MAX_SIZE < m_nCount ){
78  m_nCount = MAX_SIZE;
79  }else if( m_nCount < 0 ){
80  m_nCount = 0;
81  }
82  }
83 
84 private:
85  int m_nCount;
86  ElementType m_aElements[MAX_SIZE];
87 };
