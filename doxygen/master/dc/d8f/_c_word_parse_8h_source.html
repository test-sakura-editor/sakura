191 template< class CHAR_TYPE >
192 int CWordParse::GetWord( const CHAR_TYPE *pS, const int nLen, const CHAR_TYPE *pszSplitCharList,
193  CHAR_TYPE **ppWordStart, int *pnWordLen )
194 {
195  const CHAR_TYPE *pr = pS;
196  CHAR_TYPE *pwordstart;
197  int nwordlen;
198 
199  if( nLen < 1 ){
200  pwordstart = const_cast<CHAR_TYPE *>(pS);
201  nwordlen = 0;
202  goto end_func;
203  }
204 
205  // 区切り文字をスキップ
206  for( ; pr < pS + nLen; pr++ ){
207  // 区切り文字でない文字の間ループ
208  if( !_match_charlist(*pr, pszSplitCharList) ){
209  break;
210  }
211  }
212  pwordstart = const_cast<CHAR_TYPE*>(pr); // 単語の先頭位置を記録
213 
214  // 単語をスキップ
215  for( ; pr < pS + nLen; pr++ ){
216  // 区切り文字がくるまでループ
217  if( _match_charlist(*pr, pszSplitCharList) ){
218  break;
219  }
220  }
221  nwordlen = pr - pwordstart; // 単語の長さを記録
222 
223 end_func:
224  if( ppWordStart ){
225  *ppWordStart = pwordstart;
226  }
227  if( pnWordLen ){
228  *pnWordLen = nwordlen;
229  }
230  return pr - pS;
231 }
