66 inline int EncodeUtf8( const wchar32_t wc32, unsigned char* pDst )
67 {
68  int nret;
69 
70  if( wc32 < 0x80 ){
71  pDst[0] = static_cast<unsigned char>(wc32 & 0x00ff);
72  nret = 1;
73  }else if( wc32 < 0x800 ){
74  pDst[0] = static_cast<unsigned char>((wc32 & 0x07c0) >> 6) | 0xc0;
75  pDst[1] = static_cast<unsigned char>(wc32 & 0x003f) | 0x80;
76  nret = 2;
77  }else if( wc32 < 0x10000 ){
78  pDst[0] = static_cast<unsigned char>((wc32 & 0xf000) >> 12) | 0xe0;
79  pDst[1] = static_cast<unsigned char>((wc32 & 0x0fc0) >> 6) | 0x80;
80  pDst[2] = static_cast<unsigned char>( wc32 & 0x003f) | 0x80;
81  nret = 3;
82  }else{
83  pDst[0] = static_cast<unsigned char>((wc32 & 0x001c0000) >> 18) | 0xf0;
84  pDst[1] = static_cast<unsigned char>((wc32 & 0x0003f000) >> 12) | 0x80;
85  pDst[2] = static_cast<unsigned char>((wc32 & 0x00000fc0) >> 6) | 0x80;
86  pDst[3] = static_cast<unsigned char>( wc32 & 0x0000003f) | 0x80;
87  nret = 4;

96 inline wchar32_t DecodeUtf8( const unsigned char* pSrc, const int nSrcLen )
97 {
98  wchar32_t wc32 = 0;
99 
100  switch( nSrcLen ){
101  case 2:
102  wc32 |= static_cast<wchar32_t>(pSrc[0] & 0x1f) << 6;
103  wc32 |= static_cast<wchar32_t>(pSrc[1] & 0x3f);
104  break;
105  case 3:
106  wc32 |= static_cast<wchar32_t>(pSrc[0] & 0x0f) << 12;
107  wc32 |= static_cast<wchar32_t>(pSrc[1] & 0x3f) << 6;
108  wc32 |= static_cast<wchar32_t>(pSrc[2] & 0x3f);
109  break;
110  case 4:
111  wc32 |= static_cast<wchar32_t>(pSrc[0] & 0x07) << 18;
112  wc32 |= static_cast<wchar32_t>(pSrc[1] & 0x3f) << 12;
113  wc32 |= static_cast<wchar32_t>(pSrc[2] & 0x3f) << 6;
114  wc32 |= static_cast<wchar32_t>(pSrc[3] & 0x3f);
115  break;
116  default: // case 1:
117  wc32 = pSrc[0];
118  }
119 
120  return wc32;
121 }
