87  if( 0 == m_nIndex_Delta )
88  m_nIndex_Delta = CLogicInt(1);
89 
90  //桁増分を計算
91  if (m_pLine[m_nIndex] == WCODE::TAB){
92  if (m_tsvInfo.m_nTsvMode == TSV_MODE_TSV) {
93  m_nColumn_Delta = m_tsvInfo.GetActualTabLength(m_nColumn, m_tsvInfo.m_nMaxCharLayoutX);
94  }
95  else if (m_tsvInfo.m_nTsvMode == TSV_MODE_CSV) {
96  m_nColumn_Delta = m_nTabPadding;
97  }
98  else {
99  m_nColumn_Delta = m_nTabSpaceDx - (m_nColumn + m_nTabPadding) % m_nTabSpace;
100  }
101  } else if (m_pLine[m_nIndex] == L',' && m_tsvInfo.m_nTsvMode == TSV_MODE_CSV){
102  m_nColumn_Delta = m_tsvInfo.GetActualTabLength(m_nColumn, m_tsvInfo.m_nMaxCharLayoutX);
103  }else{
104  m_nColumn_Delta = CNativeW::GetColmOfChar( m_pLine, m_nLineLen, m_nIndex );
105  if( m_nSpacing ){
106  m_nColumn_Delta += CLayoutXInt(CNativeW::GetKetaOfChar(m_pLine, m_nLineLen, m_nIndex) * m_nSpacing);
107  }
108 // if( 0 == m_nColumn_Delta ) // 削除 サロゲートペア対策 2008/7/5 Uchi
109 // m_nColumn_Delta = CLayoutInt(1);
110  }
