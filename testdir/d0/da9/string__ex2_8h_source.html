103 template <typename T, typename ChT>
104 ptrdiff_t int2dec(
105  T value,
106  ChT* sp
107 )
108 {
109  // 符号無し整数型は対応外
110  static_assert(std::is_signed<T>::value, "T must be signed type.");
111 
112  // 一時領域
113  ChT tmp[int2dec_destBufferSufficientLength<T>()];
114  ChT *tp = tmp;
115 
116  uint8_t minAdjuster = (value == std::numeric_limits<T>::min()) ? 1 : 0;
117  // abs 関数に符号付き整数型の最小値を指定した場合の動作が未定義な事への対策
118  // 最小値だった場合は 1 加算する事で abs 関数の出力値が
119  // 符号付き整数型の最大値になるように事前に補正
120  value += minAdjuster;
121 
122  T v = abs(value);
123 
124  // 下位桁から変換する
125  do {
126  // decimal only
127  *tp++ = (ChT)('0' + (v % 10));
128  v /= 10;
129  } while (v);
130 
131  // 最小値を指定された場合に abs 関数実行前に事前に補正しているので
132  // 事後にここで最下位桁に 1 加算する事で辻褄を合わせる
133  tmp[0] += minAdjuster;
134 
135  ptrdiff_t len = tp - tmp;
136 
137  // 負の場合の符号文字
138  if (value < 0) {
139  *sp++ = '-';
140  ++len;
141  }
142 
143  // 下位桁から出力しているので文字列を逆転する
144  while (tp > tmp) {
145  --tp;
146  *sp = *tp;
147  ++sp;
148  }
149 
150  // ゼロ終端文字追加
151  *sp = '\0';
152 
153  return len;
154 }
