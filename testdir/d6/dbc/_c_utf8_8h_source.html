76 inline int CUtf8::_Utf8ToUni_char( const unsigned char* pSrc, const int nSrcLen, unsigned short* pDst, bool bCESUMode )
77 {
78  int nret;
79 
80  if( nSrcLen < 1 ){
81  return 0;
82  }
83 
84  if( bCESUMode != true ){
85  // UTF-8 の処理
86  if( nSrcLen < 4 ){
87  pDst[0] = static_cast<unsigned short>(DecodeUtf8(pSrc, nSrcLen) & 0x0000ffff);
88  nret = 1;
89  }else if( nSrcLen == 4 ){
90  // UTF-8 サロゲート領域の処理
91  wchar32_t wc32 = DecodeUtf8( pSrc, 4 );
92  EncodeUtf16Surrog( wc32, pDst );
93  nret = 2;
94  }else{
95  // 保護コード
96  pDst[0] = L'?';
97  nret = 1;
98  }
99  }else{
100  // CESU-8 の処理
101  if( nSrcLen < 4 ){
102  pDst[0] = static_cast<unsigned short>(DecodeUtf8(pSrc, nSrcLen) & 0x0000ffff);
103  nret = 1;
104  }else if( nSrcLen == 6 ){
105  // CESU-8 サロゲート領域の処理
106  pDst[0] = static_cast<unsigned short>( DecodeUtf8(&pSrc[0], 3) & 0x0000ffff );
107  pDst[1] = static_cast<unsigned short>( DecodeUtf8(&pSrc[3], 3) & 0x0000ffff );
108  nret = 2;
109  }else{
110  // 保護コード
111  pDst[0] = L'?';
112  nret = 1;
113  }
114  }
115 
116  return nret;
117 }

126 inline int CUtf8::_UniToUtf8_char( const unsigned short* pSrc, const int nSrcLen, unsigned char* pDst, bool bCESU8Mode )
127 {
128  int nret;
129 
130  if( nSrcLen < 1 ){
131  return 0;
132  }
133

135  // UTF-8 の処理
136  wchar32_t wc32;
137  if( nSrcLen == 2 ){
138  wc32 = DecodeUtf16Surrog( pSrc[0], pSrc[1] );
139  }else if( nSrcLen == 1 ){ // nSrcLen == 1
140  wc32 = pSrc[0];
141  }else{
142  wc32 = L'?';
143  }
144  nret = EncodeUtf8( wc32, &pDst[0] );
145  }else{
146  // CESU-8 の処理
147  int nclen = 0;
148  nclen += EncodeUtf8( pSrc[0], &pDst[0] );
149  if( nSrcLen == 2 ){
150  nclen += EncodeUtf8( pSrc[1], &pDst[nclen] );
151  }else{
152  ;
153  }
154  nret = nclen;
155  }
156 
157  return nret;
158 }
